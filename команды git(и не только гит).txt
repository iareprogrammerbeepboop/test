git config --global.name "имя" --global.email "почта"  - запоминает имя и почту пользователя

cd путь - переходит в нужную папку

git init - говорит гиту, что за этой папкой теперь надо следить
короче, делает гит репозиторий

git remote add origin ссылка - связывает локальный репозиторий с удалённым

git clone ссылка - скачивает все файлы из удалённого репозитория

git status - выводит все несохранённые файлов

git add файлы - подготавливает файлы к комиту (вместо файлов можно поставить точку, тогда подготовятся все файлы)

git commit -m "комент к комиту" - создаёт новый комит

git branch имя - создаёт новую ветку

git checkout имя - переходит на другую ветку
